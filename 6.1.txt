using System;

class Program
{
    static void Main()
    {
        try
        {
            Console.Write("Digite o nome do produto: ");
            string nome = Console.ReadLine();

            if (nome == "")
                throw new NullReferenceException("O nome do produto não pode estar em branco.");

            Console.Write("Digite o preço do produto: ");
            double preco = double.Parse(Console.ReadLine());

            if (preco <= 0)
                throw new Exception("O preço deve ser maior que zero.");

            Console.WriteLine($"Produto: {nome}, Preço: R${preco:F2}");
        }
        catch (FormatException)
        {
            Console.WriteLine("Erro: o preço deve ser um número válido.");
        }
        catch (NullReferenceException ex)
        {
            Console.WriteLine($"Erro: {ex.Message}");
        }
        catch (Exception ex)
        {
            Console.WriteLine($"Erro: {ex.Message}");
        }
        finally
        {
            Console.WriteLine("Bloco finally executado.");
        }
    }
}
