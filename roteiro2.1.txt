using System;

public class Produto
{
    private string nome;
    private decimal preco;

    public string Nome
    {
        get { return nome; }
    }

    public decimal Preco
    {
        get { return preco; }
        set
        {
            if (value >= 0)
            {
                preco = value;
            }
            else
            {
                Console.WriteLine("Erro: O preco nao pode ser negativo.");
            }
        }
    }

    public Produto(string nome, decimal preco)
    {
        this.nome = nome;
        Preco = preco; 
    }

    public void ExibirDetalhes()
    {
        Console.WriteLine($"Nome: {Nome}, Preco: R$ {Preco}");
    }
}

class Program
{
    static void Main()
    {
        Produto p = new Produto("PS5", 3500);
        p.ExibirDetalhes();

        p.Preco = -600; 
        p.ExibirDetalhes();
    }
}
