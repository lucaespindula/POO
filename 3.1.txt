using System;

public class Animal
{
    public string nome;

    public void EmitirSom()
    {
        Console.WriteLine("som generico do animal");
    }
}
public class Galinha : Animal
{
    public void EmitirSom()
    {
        Console.WriteLine("cococo");
    }
}
class Program
{
    static void Main()
    {
        Animal animal = new Animal();
        animal.nome = "Animal";
        animal.EmitirSom();

        Galinha galinha = new Galinha();
        galinha.nome = "Galinha";
        galinha.EmitirSom();
    }
}
