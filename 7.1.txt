using System;
using System.Collections.Generic;

public class ListaDeCompras
{
    public static void Main()
    {
        List<string> lista = new List<string>();
        string opcao = "";

        while (opcao != "5")
        {
            Console.WriteLine("\nLISTA DE COMPRAS");
            Console.WriteLine("1- Adicionar item");
            Console.WriteLine("2- Remover item");
            Console.WriteLine("3- Listar itens");
            Console.WriteLine("4- Mostrar quantidade");
            Console.WriteLine("5- Sair");
            Console.Write("Escolha: ");
            opcao = Console.ReadLine();

            if(opcao == "1")
            {
                Console.Write("Item: ");
                lista.Add(Console.ReadLine());
            }
            else if(opcao == "2")
            {
                Console.Write("Item a remover: ");
                string item = Console.ReadLine();
                if (!lista.Remove(item))
                    Console.WriteLine("Item não encontrado.");
            }
            else if(opcao == "3")
            {
                Console.WriteLine("Itens na lista:");
                foreach (string i in lista)
                    Console.WriteLine("-" + i);
            }
            else if(opcao == "4")
            {
                Console.WriteLine($"Total de itens: {lista.Count}");
            }
            else if(opcao == "5")
            {
                Console.WriteLine("Saindo...");
            }
            else
            {
                Console.WriteLine("Opção inválida.");
            }
        }
    }
}
